import { api } from "../shared/api-client.js";
import { getRole } from "../shared/role.js";
import { buildNav } from "../shared/nav-engine.js";

const navEl = document.getElementById("map-nav");
const metaEl = document.getElementById("map-meta");
const outEl = document.getElementById("map-output");
const footerStatus = document.getElementById("map-footer-status");

async function initNav() {
  const who = await api("/api/auth/whoami");
  const role = who.role || getRole();

  buildNav(navEl, {
    hub: "public-map",
    role,
    items: [{ label: "Back to Public Hub", href: "/public/index.html" }]
  });
}

async function loadMap() {
  const data = await api("/api/map/global");
  const sectors = data.sectors || [];

  metaEl.innerText = `${sectors.length} sectors`;

  outEl.innerText = JSON.stringify(data, null, 2);
}

function startHeartbeat() {
  let t = 0;
  setInterval(() => {
    footerStatus.innerText = `AI link: stable · t+${++t}s`;
  }, 1000);
}

(async function main() {
  await initNav();
  await loadMap();
  startHeartbeat();
})();
